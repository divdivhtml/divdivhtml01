}

/* 二重線ボックス（擬似要素を使用） */

}

.el-boxDoubleFrame_pseudo:before {

  padding-bottom: 2px;

  border-width: 1px;
  width: 100%;
  min-height: 100%;
  z-index: 2;
}

.el-boxDoubleFrame_pseudo.el-box03 {
  border-color: #0068b7;
}

.el-boxDoubleFrame_pseudo.el-box03:before {
  border-color: #eaf6ff;
